(f(x)\pm g(x))'=&f'(x)\pm g'(x)\notag\\
(f(x)g(x))'=&f'(x)g(x)+f(x)g'(x)\notag\\
(\dfrac{f(x)}{g(x)})'=&\dfrac{f'(x)g(x)-f(x)g'(x)}{g^2(x)}\notag\\
(f(g(x)))'=&f'(g(x))g'(x)\notag
\end{align}
