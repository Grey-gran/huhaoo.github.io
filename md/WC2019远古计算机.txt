### Subtask 4

可以往离$51\sim 100$尽可能的走，多随机几次即可

```cpp
/***************************************************************
	File name: oldcomputer4.cpp
	Author: huhao
	Create time: Sun 06 Oct 2019 10:39:50 AM CST
***************************************************************/
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
using namespace std;
#define fr(i,a,b) for(int i=(a),end_##i=(b);i<=end_##i;i++)
#define fd(i,a,b) for(int i=(a),end_##i=(b);i>=end_##i;i--)
int read()
{
	int r=0,t=1,c=getchar();
	while(c<'0'||c>'9')
	{
		t=c=='-'?-1:1;
		c=getchar();
	}
	while(c>='0'&&c<='9')
	{
		r=r*10+c-48;
		c=getchar();
	}
	return r*t;
}
#include<vector>
const int N=1010;
vector<int>e[N];
int n,m,d[N],q[N],l,r,pre[N],a[N],p[N],P[N];
char ans[N][50][20];
#define fp(...) fprintf(stderr,__VA_ARGS__)
#define add(s,...) sprintf(s,__VA_ARGS__)
#include<stdlib.h>
int main()
{
	freopen("oldcomputer4.in","r",stdin);
	freopen("oldcomputer4.out","w",stdout);
	read();
	n=read();
	m=read();
	fr(i,1,n)
		e[i].push_back(i);
	fr(i,1,m)
	{
		int u=read(),v=read();
		e[u].push_back(v);
		e[v].push_back(u);
	}
	fr(i,1,n)
		d[i]=n+2;
	fr(i,51,100)
	{
		q[++r]=i;
		d[i]=1;
	}
	l=1;
	while(l<=r)
	{
		int u=q[l];
		l++;
		for(auto v:e[u])
			if(d[v]>d[u]+1)
			{
				d[v]=d[u]+1;
				pre[v]=u;
				q[++r]=v;
			}
	}
	unsigned long long sed=0;
	while(1)
	{
		memset(ans,0,sizeof(ans));
		srand(++sed);
		int s=50,l=50,cl=1;
		fr(i,1,s)
		{
			add(ans[i][1],"read 0 a");
			a[i]=i;
			P[i]=1;
		}
		while(l)
		{
			random_shuffle(a+1,a+s+1);
			cl++;
			fr(i,1,n)
				p[i]=0;
			fr(i,1,s)
				if(a[i]&&!p[a[i]])
				{
					int Pre=pre[a[i]];
					if(p[Pre]||P[Pre])
					{
						Pre=-1;
						for(auto v:e[a[i]])
							if(d[v]<=d[a[i]]&&!p[v]&&!P[v])
							{
								if(!Pre||rand()%3==0)
									Pre=v;
							}
					}
					if(Pre>=0&&!p[Pre]&&!P[Pre])
					{
						add(ans[a[i]][cl],"write a %d",Pre);
						add(ans[Pre][cl],"read %d a",a[i]);
						p[a[i]]=p[Pre]=1;
						P[a[i]]=0;
						P[Pre]=1;
						if(Pre==0)
							l--;
						a[i]=Pre;
						p[0]=P[0]=0;
					}
				}
		}
		fp("%d %llu\n",cl,sed);
		if(cl<=7)
		{
			fr(i,1,n)
			{
				printf("node %d\n",i);
				fr(j,1,cl)
				{
					if(strlen(ans[i][j])==0)
						add(ans[i][j],"add a 0");
					printf("%s\n",ans[i][j]);
				}
			}
			return 0;
		}
	}
	return 0;
}
```
